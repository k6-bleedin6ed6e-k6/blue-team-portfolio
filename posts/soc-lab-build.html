DET-001
DNS Tunneling
MITRE T1071.004  ·  HIGH  ·  index=nextdns
index=nextdns | eval domain_length=len(domain), subdomain_length=len(mvindex(split(domain,"."),0)) | where domain_length > 50 OR subdomain_length > 30 | bin _time span=5m | stats count as queries, dc(domain) as unique_domains by _time, client_ip | where queries > 20 OR unique_domains > 15 | eval severity=case(queries>50,"CRITICAL",queries>20,"HIGH",1=1,"MEDIUM") | table _time, client_ip, queries, unique_domains, severity
● SPL — Splunk Cloud (Phase 1) · Sigma — DET-001_dns_tunneling.yml
DET-002
DGA Detection (Domain Generation Algorithm)
MITRE T1568.002  ·  HIGH  ·  index=nextdns
index=nextdns | eval consonants=len(replace(lower(domain),"[aeiou.]","")) | eval vowels=len(replace(lower(domain),"[^aeiou]","")) | eval ratio=consonants/max(vowels,1) | where ratio > 4 AND len(domain) > 40 | sort - len(domain)
● SPL — Splunk Cloud (Phase 1)
DET-003
DNS Beaconing
MITRE T1071.004  ·  MEDIUM  ·  index=nextdns
index=nextdns | sort _time | delta _time as gap | stats avg(gap) as avg_gap, count by domain, client_ip | where avg_gap > 300 AND avg_gap < 3600 AND count > 10 | eval beacon_interval=round(avg_gap/60,1)." min" | table domain, client_ip, count, beacon_interval
● SPL — Splunk Cloud (Phase 1)
DET-004
SSH Brute Force
MITRE T1110  ·  HIGH  ·  index=linux
index=linux sourcetype=linux_secure | rex "Invalid user (?<user>\S+) from (?<src_ip>\d+\.\d+\.\d+\.\d+)" | where isnotnull(src_ip) | bin _time span=5m | stats count as attempts, values(user) as usernames by _time, src_ip | where attempts > 5 | eventstats sum(attempts) as total by src_ip | where total > 20 | table _time, src_ip, usernames, attempts, total | sort - total
● SPL — Splunk Cloud (Phase 1) · Live data on AWS EC2 honeypot now
DET-005
Firewall Port Scan
MITRE T1046  ·  MEDIUM  ·  index=linux
index=linux sourcetype=ufw | rex "SRC=(?<src_ip>\d+\.\d+\.\d+\.\d+).*DPT=(?<dst_port>\d+)" | bin _time span=1m | stats dc(dst_port) as ports_scanned, values(dst_port) as ports by _time, src_ip | where ports_scanned > 10 | sort - ports_scanned
● SPL — Splunk Cloud (Phase 1)
DET-006
Suspicious Process Creation (Office → Shell)
MITRE T1059  ·  HIGH  ·  index=windows · Sysmon EventCode=1
index=windows sourcetype="XmlWinEventLog:...Sysmon" EventCode=1 | rex field=Message "CommandLine: (?<cmdline>[^\r\n]+)" | rex field=Message "ParentImage: (?<parent>[^\r\n]+)" | where match(cmdline,"(?i)(powershell|cmd|wscript|mshta|rundll32)") AND match(parent,"(?i)(word|excel|outlook|winword)") | table _time, ComputerName, cmdline, parent
● SPL — Splunk Cloud + Sysmon (Phase 2)
DET-007
Credential Dumping / LSASS Access
MITRE T1003  ·  CRITICAL  ·  index=windows · Sysmon EventCode=10
index=windows sourcetype="XmlWinEventLog:...Sysmon" EventCode=10 | rex field=Message "TargetImage: (?<target>[^\r\n]+)" | rex field=Message "CallTrace: (?<calltrace>[^\r\n]+)" | where match(target,"(?i)lsass\.exe") AND match(calltrace,"(?i)dbghelp\.dll|dbgcore\.dll|unknown") | table _time, ComputerName, source, target, calltrace
● SPL — Splunk Cloud + Sysmon (Phase 2)
DET-008
PowerShell Encoded Command
MITRE T1059.001  ·  HIGH  ·  index=windows · Sysmon EventCode=1
index=windows sourcetype="XmlWinEventLog:...Sysmon" EventCode=1 | rex field=Message "CommandLine: (?<cmdline>[^\r\n]+)" | where match(cmdline,"(?i)(-enc|-encodedcommand|-e\s+[A-Za-z0-9+/=]{20,})") | rex field=cmdline "(?i)(?:-enc|-encodedcommand|-e)\s+(?<encoded>[A-Za-z0-9+/=]+)" | table _time, ComputerName, cmdline, encoded

New Local Admin Account Created
